Given an array of intervals where each interval is represented as [start, end]. 
Each pair denotes the start and end of an interval on a number line. 

For example, [1, 3] represents an interval that starts at 1 and ends at 3.
Merge all overlapping intervals in the array and return an array of non-overlapping intervals that fully cover the ranges specified by the input intervals.

```jsx 
function mergeOverlappingIntervals(iv) {
 let s = iv.sort((a,b)=>a[0]-b[0]);
 let m = [s[0]];  
 for (let i = 1; i < s.length; i++) {
    let curr = s[i];
    let lm = m[m.length-1];
     // if current start falls into the last merged interval 
    if(curr[0] <= lm[1]){
     // check and update the end of last merged interval       
        lm[1] = Math.max(lm[1], curr[1]);  
    } else {
        m.push(curr)
    } 
 }
    return m;
}
```
