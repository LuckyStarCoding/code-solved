Given an array of meeting time intervals where each element represents a meeting time as [start, end], 
determine whether a person can attend all the meetings without any overlap between meeting times.
A 2D array of integers intervals of size n x 2, where n is the number of meetings and each interval [start, end] represents a meeting starting at time start and ending at time end

```jsx 
function isMeetingCalendarValid(iv) {
  if (iv.length < 2) {
    return true;
  }
  let s = iv.sort((a, b) => a[0] - b[0]);
  for (let i = 0; i < s.length - 1; i++) {
    if (s[i][1] > s[i + 1][0]) {
      return false;
    }
  }
  return true;
}
```
