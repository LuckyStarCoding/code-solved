You are given two strings s and t.
String t is generated by randomly shuffling string s and then adding one more letter at a random position.
Return the letter that was added to t.

<TabItem value="solution1" label="Solution 1" >
```jsx 
function findTheDifference (s, t) {
let sh = [...s].sort();
let sl = [...t].sort();
 for (let i = 0; i < sh.length; i++) {
    if(sh[i]!==sl[i]){return sl[i]}
 }
	return sl[sl.length - 1]
}
```

```jsx

 function findTheDifference (s, t) {
    let counts = {};

    // 1. Efficiently build the frequency map for the shorter string (s)
    for (const char of s) {
        // Increment the count for each character encountered
        counts[char] = (counts[char] || 0) + 1;
    }

    // 2. Iterate through the longer string (t) and find the mismatch
    for (const char of t) {
        // The difference is found if:
        // a. The character is NOT in the map (counts[char] is undefined)
        // OR
        // b. The character IS in the map, but its count has already dropped to 0
        if (!counts[char]) { 
            return char; // FOUND IT!
        }
        
        // If found, decrement the count and continue
        counts[char]--;
    }
}
```
