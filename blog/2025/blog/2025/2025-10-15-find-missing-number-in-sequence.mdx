Given an array numbers of size `n` storing n different integers which fall within the range `[0, n]`, implement a function to identify the missing element in the array. 
All numbers except one are present in the array. Find the missing number.

```jsx
 function findMissingNumberInSequence(nums) {
  let s = nums.sort((a,b)=>a-b);
      for (let i = 0; i < s.length+1; i++) {
          if(s[i]!==i) {return i} 
      }
}
```

<TabItem value="solution2" label="Solution 2 (Optimal)" >
```jsx 
function findMissingNumberOptimal(nums) {
    const N = nums.length;
    
    // 1. Calculate the expected sum of the complete sequence (0 to N).
    // Formula: N * (N + 1) / 2
    const expectedSum = N * (N + 1) / 2;
    
    // 2. Calculate the actual sum of the given array.
    let actualSum = 0;
    for (const num of nums) {
        actualSum += num;
    }
    
    // 3. The missing number is the difference.
    return expectedSum - actualSum;
}

```
